154 template <typename T>
155 class IkSolution : public IkSolutionBase<T>
156 {
157 public:
158  IkSolution(const std::vector<IkSingleDOFSolutionBase<T> >& vinfos, const std::vector<int>& vfree) {
159  _vbasesol = vinfos;
160  _vfree = vfree;
161  }
162 
163  virtual void GetSolution(T* solution, const T* freevalues) const {
164  for(std::size_t i = 0; i < _vbasesol.size(); ++i) {
165  if( _vbasesol[i].freeind < 0 )
166  solution[i] = _vbasesol[i].foffset;
167  else {
168  solution[i] = freevalues[_vbasesol[i].freeind]*_vbasesol[i].fmul + _vbasesol[i].foffset;
169  if( solution[i] > T(3.14159265358979) ) {
170  solution[i] -= T(6.28318530717959);
171  }
172  else if( solution[i] < T(-3.14159265358979) ) {
173  solution[i] += T(6.28318530717959);
174  }
175  }
176  }
177  }
178 
179  virtual void GetSolution(std::vector<T>& solution, const std::vector<T>& freevalues) const {
180  solution.resize(GetDOF());
181  GetSolution(&solution.at(0), freevalues.size() > 0 ? &freevalues.at(0) : NULL);
182  }
183 
184  virtual const std::vector<int>& GetFree() const {
185  return _vfree;
186  }
187  virtual const int GetDOF() const {
188  return static_cast<int>(_vbasesol.size());
189  }
190 
191  virtual void Validate() const {
192  for(size_t i = 0; i < _vbasesol.size(); ++i) {
193  if( _vbasesol[i].maxsolutions == (unsigned char)-1) {
194  throw std::runtime_error("max solutions for joint not initialized");
195  }
196  if( _vbasesol[i].maxsolutions > 0 ) {
197  if( _vbasesol[i].indices[0] >= _vbasesol[i].maxsolutions ) {
198  throw std::runtime_error("index >= max solutions for joint");
199  }
200  if( _vbasesol[i].indices[1] != (unsigned char)-1 && _vbasesol[i].indices[1] >= _vbasesol[i].maxsolutions ) {
201  throw std::runtime_error("2nd index >= max solutions for joint");
202  }
203  }
204  if( !std::isfinite(_vbasesol[i].foffset) ) {
205  throw std::runtime_error("foffset was not finite");
206  }
207  }
208  }
209 
210  virtual void GetSolutionIndices(std::vector<unsigned int>& v) const {
211  v.resize(0);
212  v.push_back(0);
213  for(int i = (int)_vbasesol.size()-1; i >= 0; --i) {
214  if( _vbasesol[i].maxsolutions != (unsigned char)-1 && _vbasesol[i].maxsolutions > 1 ) {
215  for(size_t j = 0; j < v.size(); ++j) {
216  v[j] *= _vbasesol[i].maxsolutions;
217  }
218  size_t orgsize=v.size();
219  if( _vbasesol[i].indices[1] != (unsigned char)-1 ) {
220  for(size_t j = 0; j < orgsize; ++j) {
221  v.push_back(v[j]+_vbasesol[i].indices[1]);
222  }
223  }
224  if( _vbasesol[i].indices[0] != (unsigned char)-1 ) {
225  for(size_t j = 0; j < orgsize; ++j) {
226  v[j] += _vbasesol[i].indices[0];
227  }
228  }
229  }
230  }
231  }
